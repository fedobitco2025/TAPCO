async function withdrawTokens(amount) {
    if (!web3 || !userAddress) {
        infoBox.textContent = "يجب عليك ربط المحفظة أولاً.";
        return;
    }

    try {
        const contract = new web3.eth.Contract(CONTRACT_ABI, CONTRACT_ADDRESS);

        const decimals = await contract.methods.decimals().call();
        const realAmount = web3.utils.toBN(amount).mul(
            web3.utils.toBN(10).pow(web3.utils.toBN(decimals))
        );

        infoBox.textContent = "جاري تنفيذ عملية السحب...";

        await contract.methods.transfer(userAddress, realAmount).send({ from: userAddress });

        infoBox.textContent = "🎉 تم السحب بنجاح! تم إرسال TAPCO إلى محفظتك.";
        playSound(successSound);
        vibrate(80);

    } catch (err) {
        console.error(err);
        infoBox.textContent = "فشل السحب. تأكد من وجود رصيد كافٍ في العقد.";
    }
}

async function connectWallet() {
    if (window.ethereum) {
        try {
            // طلب الإذن من المستخدم

            userAddress = accounts[0];

            localStorage.setItem("myReferral", userAddress);

            // مكافأة الإحالة
            const referredBy = localStorage.getItem("referredBy");

            if (referredBy && !localStorage.getItem("refRewardGiven")) {
                score += 200; // مكافأة اللاعب الجديد

    function setupDoubleClickTask(el) {
      let clickCount = 0;
      let firstClickTime = 0;
      const windowMs = 450;

      el.addEventListener('click', () => {
        const now = Date.now();
        if (clickCount === 0) {
          clickCount = 1;
          firstClickTime = now;
        } else {
          if (now - firstClickTime <= windowMs) {
            completeBonus(el);
          }
          clickCount = 0;
        }
      });
    }

    function setupLongPressTask(el) {
      let pressTimer = null;
      const requiredMs = 800;

      const start = () => {
        pressTimer = setTimeout(() => {
          completeBonus(el);
        }, requiredMs);
      };
      const cancel = () => {
        if (pressTimer) {
          clearTimeout(pressTimer);
          pressTimer = null;
        }
      };

      el.addEventListener('mousedown', start);
      el.addEventListener('mouseup', cancel);
      el.addEventListener('mouseleave', cancel);

      el.addEventListener('touchstart', start);
      el.addEventListener('touchend', cancel);
      el.addEventListener('touchcancel', cancel);
    }

    function startBoost(durationSeconds) {
      boostActive = true;
      boostRemaining = durationSeconds;
      tapPower = baseTapPower * 1.5;
      updateBoostUI();

      if (boostIntervalId) clearInterval(boostIntervalId);
      boostIntervalId = setInterval(() => {
        boostRemaining--;
        if (boostRemaining <= 0) {
          boostActive = false;
          tapPower = baseTapPower;
          clearInterval(boostIntervalId);
          boostIntervalId = null;
        }
        updateBoostUI();
      }, 1000);
    }

      if (tapArea.contains(el)) tapArea.removeChild(el);

function buyUpgrade(key) {
    const up = upgrades[key];
    if (!up) return;

    if (score < up.cost) {
        infoBox.textContent = "نقاطك غير كافية لهذه الترقية.";
        return;
    }

    // خصم السعر
    score -= up.cost;

    // رفع المستوى
    up.level += 1;

    // زيادة السعر في كل مرة (تصاعدي)
    up.cost = Math.floor(up.cost * 1.8);

    // تأثيرات خاصة لكل ترقية
    if (key === "maxEnergy") {
        maxEnergy += up.baseBonus;
    }

    if (key === "energyRegen") {
        energyRegenRate += up.baseBonus; // لو عندك متغير آخر، أرسله لي
    }

    updateUI();
    infoBox.textContent = "تمت الترقية بنجاح!";
    playSound(successSound);
    saveUpgrades();
