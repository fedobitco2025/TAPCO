    function updateEventProgress(type, value) {
      if (!Number.isFinite(value) || value <= 0) return;
      updateActiveEvents();

      events.forEach(function(evt) {
        if (!evt.isActive) return;
        if (evt.type !== type) return;

        const progressObj = playerEventsProgress[evt.id];
        if (!progressObj || progressObj.isClaimed || progressObj.isCompleted) return;

        progressObj.progress += value;
        if (progressObj.progress >= evt.target) {
          progressObj.progress = evt.target;
          progressObj.isCompleted = true;
        }
      });

      saveEventsData();
    }

    function grantEventReward(reward) {
      if (!reward) return;

      if (reward.type === 'xp') {
        addXP(reward.value || 0);
      } else if (reward.type === 'token') {
        score += (reward.value || 0);
        daily += (reward.value || 0);
        totalPointsEarned += (reward.value || 0);
        checkTotalPointsMilestones();
      } else if (reward.type === 'boost') {
        addBoostToEventInventory(reward.id || 'multiplier', reward.amount || 1);
      } else if (reward.type === 'skin') {
        const skinId = reward.id || 'default-skin';
        if (!eventInventory.skins.includes(skinId)) {
          eventInventory.skins.push(skinId);
        }
      } else if (reward.type === 'box') {
        eventInventory.boxes += reward.amount || 1;
        if (eventInventory.boxes > 0) {
          eventInventory.boxes--;
          openEventBox();
          return;
        }
      }

      updateUI();
      savePlayerData();
      saveEventsData();
    }

    function formatDuration(ms) {
      const totalSeconds = Math.max(0, Math.floor(ms / 1000));
      const hours = Math.floor(totalSeconds / 3600);
      const minutes = Math.floor((totalSeconds % 3600) / 60);
      const seconds = totalSeconds % 60;
      if (hours > 0) return `${hours}س ${minutes}د`;
      if (minutes > 0) return `${minutes}د ${seconds}ث`;
      return `${seconds}ث`;
    }

    function getUpgradeCost(key, targetLevel) {
      const level = Math.max(1, targetLevel);
      const baseCostMap = {
        tapPower: 20,
        maxEnergy: 30,
        energyRegen: 40,
        autoTap: 50,
        superAutoTap: 75
      };
      const baseCost = baseCostMap[key] || 30;
      return Math.floor(baseCost * Math.pow(economyUpgradeGrowthRate, Math.max(0, level - 1)));
    }

    function getUpgradeTotalCostToLevel(key, currentLevel, targetLevel) {
      if (targetLevel <= currentLevel) return 0;
      let total = 0;
      for (let lvl = currentLevel + 1; lvl <= targetLevel; lvl++) {
        total += getUpgradeCost(key, lvl);
      }
      return total;
    }

    function getUpgradeMilestoneText(key, currentLevel, targetLevel) {
      const total = getUpgradeTotalCostToLevel(key, currentLevel, targetLevel);
      if (total <= 0) return `تم الوصول إلى مستوى ${targetLevel}`;
      return `${total.toLocaleString('en-US')} نقطة للوصول إلى مستوى ${targetLevel}`;
    }

    function addXP(amount) {

      const boostedAmount = amount * getXpBoostMultiplier() * getPrestigeBonusMultiplier('xp');
      const finalXpGain = boostedAmount * tierXpMultiplier;
      xp += finalXpGain;
  dailyXP += finalXpGain;

    function getXPRequiredForLevel(lvl) {
      if (lvl < xpRequirements.length) {
        return xpRequirements[lvl];
      }

      // Keep post-20 progression slow with gradually increasing increments.
      const overflowLevel = lvl - 20;
      const base = xpRequirements[xpRequirements.length - 1];
      return base + (overflowLevel * (2600 + overflowLevel * 220));
    }

    // ===== HYBRID BOT DETECTION SYSTEM =====
    function getCurrentCps(now = Date.now(), windowMs = 1000) {
      const start = now - windowMs;
      let count = 0;
      for (let i = tapTimestamps.length - 1; i >= 0; i--) {
        if (tapTimestamps[i] < start) break;
        count++;
      }
      return count / (windowMs / 1000);
    }

    function trackTapSpeed() {
      const now = Date.now();
      tapTimestamps.push(now);
      const cutoff = now - 15000;
      while (tapTimestamps.length && tapTimestamps[0] < cutoff) tapTimestamps.shift();

      // Signal 1: sustained fast tapping using a 3-second window for stability
      // Max CPS with 140ms cooldown ≈ 7.14, threshold set to 5.5 to be safely detectable
      const cps = getCurrentCps(now, 3000);
      if (cps > 5.5) {
        if (!highSpeedStartAt) highSpeedStartAt = now;
        if (now - highSpeedStartAt >= 10000) signalHighSpeedDetected = true;
      } else {
        highSpeedStartAt = 0;
      }

      // Signal 2: 80% of intervals within 20ms of median over 5 seconds
      // FIX: use 4000ms span threshold (not 5000) since window is filtered to last 5000ms
      const windowStart = now - 5000;
      const recent = tapTimestamps.filter((t) => t >= windowStart);
      if (recent.length >= 12) {
        const intervals = [];
        for (let i = 1; i < recent.length; i++) intervals.push(recent[i] - recent[i - 1]);
        const sorted = [...intervals].sort((a, b) => a - b);
        const median = sorted[Math.floor(sorted.length / 2)];
        const closeCount = intervals.filter(iv => Math.abs(iv - median) < 20).length;
        const ratio = closeCount / intervals.length;
        if (ratio >= 0.8 && (recent[recent.length - 1] - recent[0]) >= 4000) {
          signalStableIntervalDetected = true;
        }
      }

      // Signal 3 completion: after reward click, rapid tapping resumes quickly
      if (awaitingRapidRestart) {
        if (now - rewardHandledAt > 10000) {
          awaitingRapidRestart = false;
          hybridCycleCount = 0;
        } else if (cps > 6) {
          hybridCycleCount++;
          awaitingRapidRestart = false;
          if (hybridCycleCount >= 3) signalHybridPatternDetected = true;
        }
      }
      // Immediate tier promotion: if score is already detectable, apply instantly
      const instantScore = calculateBehaviorScore();
      if (instantScore >= 7 && playerTier !== 'C') {
        setTierWithHold('C', true);
        enforceTierEffects();
        savePlayerData();
        updateUI();
      } else if (instantScore >= 4 && playerTier !== 'C') {
        setTierWithHold('B', true);
        savePlayerData();
        updateUI();
      }
    }

    function setTierWithHold(newTier, showWarning = false) {
      const now = Date.now();
      const previousTier = playerTier;

      // Anti-loop escalation: after a prior B sanction, next B-level detection becomes C directly.
      if (newTier === 'B' && hadTierBViolation) {
        newTier = 'C';
      }

      if (newTier === 'A') {
        playerTier = 'A';
        tierLockedUntil = 0;
      } else if (newTier === 'B') {
        playerTier = 'B';
        tierLockedUntil = Math.max(tierLockedUntil, now + tierHoldDurationMs.B);
        hadTierBViolation = true;
      } else if (newTier === 'C') {
        playerTier = 'C';
        tierLockedUntil = Math.max(tierLockedUntil, now + tierHoldDurationMs.C);
        if (previousTier !== 'C') {
          cViolationCount++;
        }
        if (cViolationCount >= 2) {
          isPermanentlyBanned = true;
        }
      }

      const tierChanged = previousTier !== playerTier;
      if (showWarning && (newTier === 'B' || newTier === 'C') && (tierChanged || previousTier === newTier)) {
        triggerBotWarning(playerTier);
      }

      return tierChanged;
    }

    function isCurrentTierLocked(now = Date.now()) {
      return (playerTier === 'B' || playerTier === 'C') && now < tierLockedUntil;
    }

    function markChallengeResolved(success) {
      if (!success) {
        if (hybridCycleCount > 0) hybridCycleCount = 0;
        awaitingRapidRestart = false;
        return;
      }

      const now = Date.now();
      const reactionDelay = challengeShownAt ? (now - challengeShownAt) : 0;
      const looksLikeStopAndManualReward = challengeHadRapidBefore && reactionDelay >= 500 && reactionDelay <= 8000;

      if (looksLikeStopAndManualReward) {
        awaitingRapidRestart = true;
        rewardHandledAt = now;
      } else {
        hybridCycleCount = 0;
        awaitingRapidRestart = false;
      }
    }

    function calculateBehaviorScore() {
      let score = 0;
      if (signalHighSpeedDetected) score += 2;
      if (signalStableIntervalDetected) score += 2;
      if (signalHybridPatternDetected) score += 3;
      return score;
    }

    function enforceTierEffects() {
      if (playerTier === 'C' && activeBoosts.autoTap.active) {
        activeBoosts.autoTap.active = false;
        stopAutoTapTracking('autoTap');
        if (activeBoosts.autoTap.intervalId) {
          clearInterval(activeBoosts.autoTap.intervalId);
          activeBoosts.autoTap.intervalId = null;
        }
      }
      if (playerTier === 'C' && activeBoosts.superAutoTap.active) {
        activeBoosts.superAutoTap.active = false;
        stopAutoTapTracking('superAutoTap');
        if (activeBoosts.superAutoTap.intervalId) {
          clearInterval(activeBoosts.superAutoTap.intervalId);
          activeBoosts.superAutoTap.intervalId = null;
        }
      }
    }

    function resetBehaviorWindow() {
      signalHighSpeedDetected = false;
      signalStableIntervalDetected = false;
      signalHybridPatternDetected = false;
      highSpeedStartAt = 0;
      hybridCycleCount = 0;
      awaitingRapidRestart = false;
      rewardHandledAt = 0;
    }

    function updatePlayerTier() {
      const now = Date.now();
      if (now - lastBehaviorCheckTime < behaviorCheckIntervalMs) return;

      lastBehaviorCheckTime = now;
      behaviorScore = calculateBehaviorScore();

      let newTier = 'A';
      if (behaviorScore >= 7) newTier = 'C';
      else if (behaviorScore >= 4) newTier = 'B';

      // Keep player in B/C until hold timer expires.
      if (isCurrentTierLocked(now)) {
        if (newTier === 'C' && playerTier !== 'C') {
          setTierWithHold('C', true);
        } else if (newTier === playerTier && (newTier === 'B' || newTier === 'C')) {
          tierLockedUntil = Math.max(tierLockedUntil, now + tierHoldDurationMs[newTier]);
          triggerBotWarning(playerTier);
        }
        enforceTierEffects();
        savePlayerData();
        resetBehaviorWindow();
        updateUI();
        return;
      }

      if (newTier !== playerTier) {
        setTierWithHold(newTier, newTier === 'B' || newTier === 'C');
        const tierMessages = {
          A: '✅ الفئة A - نشاط طبيعي',
          B: '⚠️ الفئة B - نشاط غير طبيعي',
          C: '🔴 الفئة C - نشاط آلي مرتفع الخطورة'
        };
        infoBox.textContent = tierMessages[playerTier];
      }

      enforceTierEffects();
      savePlayerData();
      resetBehaviorWindow();
      updateUI();
    }

    function updateMaxEnergy() {
      const energyLevel = upgrades.maxEnergy.level;
      maxEnergy = Math.floor(100 + (energyLevel * 12));
      if (energy > maxEnergy) energy = maxEnergy;
    }

    function getAutoTapInterval() {
      const baseInterval = 1000; // 1 second
      const levelModifier = Math.max(0.5, 1 - (level - 1) * 0.005); // +0.5% speed per level
      return baseInterval * levelModifier;
    }

    function getEnergyRegenRate() {
      const regenLevel = upgrades.energyRegen.level;
      let rate = 1 + (regenLevel * economyEnergyRegenGrowthRate);
      
      // Apply Daily Boost multiplier
      const dailyBoostMultiplier = updateDailyBoostMultiplier();
      rate *= dailyBoostMultiplier;
      
      // Apply Temp Boost multiplier (for regen specifically)
      if (dailyState.tempBoost.active && dailyState.tempBoost.type === 'regen') {
        rate *= dailyState.tempBoost.value;
      }
      
      return rate;
    }

    function getDailyRewardValue() {
      const base = 100 + (level * 15);
      const tapContribution = Math.sqrt(Math.max(1, getTapPower())) * 35;
      const prestigeDailyMultiplier = 1 + (prestigeLevel * 0.02);
      return Math.floor((base + tapContribution) * prestigeDailyMultiplier);
    }

    function applyOfflineEnergyRegen() {
      updateMaxEnergy();
      const now = Date.now();
      if (!lastEnergyTimestamp) lastEnergyTimestamp = now;
      const elapsedSec = Math.max(0, (now - lastEnergyTimestamp) / 1000);
      const regenAmount = elapsedSec * getEnergyRegenRate();
      energy = Math.min(maxEnergy, energy + regenAmount);
      lastEnergyTimestamp = now;
    }

    function saveUpgrades() {
      if (!centralSaveInProgress) {
        saveGame();

      localStorage.setItem("tapco_upgrades", JSON.stringify(upgrades));
    }

    function startAutoTapTracking(type) {
      const now = Date.now();

      if (type === 'autoTap') {
        autoTapUses++;
        autoTapStartTime = now;
      } else if (type === 'superAutoTap') {
        superAutoTapUses++;
        superAutoTapStartTime = now;
      } else {
        return;
      }

      savePlayerData();
      saveAchievements();
      checkAllAchievements();
    }

    function stopAutoTapTracking(type) {
      const now = Date.now();
      let didUpdate = false;

      if (type === 'autoTap' && autoTapStartTime > 0) {
        autoTapTotalTime += Math.max(0, now - autoTapStartTime);
        autoTapStartTime = 0;
        didUpdate = true;
      }

      if (type === 'superAutoTap' && superAutoTapStartTime > 0) {
        superAutoTapTotalTime += Math.max(0, now - superAutoTapStartTime);
        superAutoTapStartTime = 0;
        didUpdate = true;
      }

      if (didUpdate) {
        savePlayerData();
        saveAchievements();
        checkAllAchievements();
      }
    }

    function checkBoosts() {
      const now = Date.now();
      Object.keys(activeBoosts).forEach((key) => {
        const boost = activeBoosts[key];
        if (boost.active && boost.endTime && now > boost.endTime) {
          boost.active = false;
          if (key === 'autoTap') {
            stopAutoTapTracking('autoTap');

    function classifyPlayer() {

      const today = now.toDateString();
      if (lastClassificationDate !== today) {
        // Update consecutive days
        const yesterday = new Date(now);
        yesterday.setDate(yesterday.getDate() - 1);
        if (lastClassificationDate === yesterday.toDateString()) {
          consecutiveDays++;
        } else {
          consecutiveDays = 1;
        }

        // Reset daily stats only (tier is behavior-driven and updated every 5 minutes)
        dailyClicks = 0;
        dailyPoints = 0;
        sessionTime = 0;
        lastClassificationDate = today;
        savePlayerData();
      }
    }

    function getTierMultiplier() {
      // Points penalty: A: 100% | B: 30% reduction | C: 70% reduction
      const scoreMultipliers = { A: 1.0, B: 0.7, C: 0.3 };
      return scoreMultipliers[playerTier] || 1.0;

    function getScoreBoostMultiplier() {
      let multiplier = 1;
      if (activeBoosts.multiplier.active) multiplier = Math.max(multiplier, activeBoosts.multiplier.multiplier);
      if (activeBoosts.megaMultiplier.active) multiplier = Math.max(multiplier, activeBoosts.megaMultiplier.multiplier);
      return multiplier;
    }

    function getXpBoostMultiplier() {
      let multiplier = 1;
      if (activeBoosts.xpMultiplier.active) multiplier = Math.max(multiplier, activeBoosts.xpMultiplier.multiplier);
      if (activeBoosts.xpMegaMultiplier.active) multiplier = Math.max(multiplier, activeBoosts.xpMegaMultiplier.multiplier);
      return multiplier;
    }

    function getPrestigeBonusMultiplier(type) {
      if (type === 'points') return (permanentBonuses.pointsMultiplier || 1);
      if (type === 'xp') return (permanentBonuses.xpMultiplier || 1);
      if (type === 'energy') return (permanentBonuses.energyMultiplier || 1);
      return 1;
    }

    function recalculateReferralTapPowerBonus() {
      const cappedReferrals = Math.max(0, Math.min(20, referralCount));
      if (cappedReferrals === 0) {
        referralTapPowerBonusPercent = 0;
        return;
      }
      referralTapPowerBonusPercent = 0.05 + Math.max(0, cappedReferrals - 1) * 0.01;
    }

    function calculateTapPowerRaw() {
      const tapUpgradeLevel = upgrades.tapPower.level;
      let power = economyBaseTap * Math.pow(economyTapGrowthRate, tapUpgradeLevel);
      const tapPowerBonusMultiplier = 1 + (achievementBonuses.tapPowerPercent || 0) + referralTapPowerBonusPercent;
      power *= tapPowerBonusMultiplier;
      power *= getScoreBoostMultiplier();
      power *= getPrestigeBonusMultiplier('points');
      power *= getTierMultiplier();
      return power;
    }

    function refreshEconomyCache() {
      cachedTapPower = calculateTapPowerRaw();
      cachedAutoTapPower = cachedTapPower * 0.35;
      cachedSuperAutoTapPower = cachedTapPower * 1.5;

      cachedUpgradeCosts.tapPower = getUpgradeCost('tapPower', upgrades.tapPower.level + 1);
      cachedUpgradeCosts.maxEnergy = getUpgradeCost('maxEnergy', upgrades.maxEnergy.level + 1);
      cachedUpgradeCosts.energyRegen = getUpgradeCost('energyRegen', upgrades.energyRegen.level + 1);
      cachedUpgradeCosts.autoTap = getUpgradeCost('autoTap', autoTapUses + 1);
      cachedUpgradeCosts.superAutoTap = getUpgradeCost('superAutoTap', superAutoTapUses + 1);
    }

    function getTapPower() {
      if (cachedTapPower === null) {
        refreshEconomyCache();
      }
      
      let power = cachedTapPower;
      
      // Apply Daily Boost multiplier
      const dailyBoostMultiplier = updateDailyBoostMultiplier();
      power *= dailyBoostMultiplier;
      
      // Apply Temp Boost multiplier (for tap power specifically)
      if (dailyState.tempBoost.active && dailyState.tempBoost.type === 'tap') {
        power *= dailyState.tempBoost.value;
      }
      
      return power;
    }

    const audioCtx = new (window.AudioContext || window.webkitAudioContext)();

    function playTone(frequency, duration = 120, type = 'sine', volume = 0.15) {
      if (!audioCtx) return;
      if (audioCtx.state === 'suspended') {
        audioCtx.resume().catch(() => {});
      }
      const oscillator = audioCtx.createOscillator();
      const gain = audioCtx.createGain();
      oscillator.type = type;
      oscillator.frequency.value = frequency;
      gain.gain.value = volume;
      oscillator.connect(gain);
      gain.connect(audioCtx.destination);
      oscillator.start();
      oscillator.stop(audioCtx.currentTime + duration / 1000);
      oscillator.onended = () => {
        if (gain) gain.disconnect();
        if (oscillator) oscillator.disconnect();
      };
    }

    let lastRegenCheck = Date.now();

    function tickEnergyRegen() {
      const now = Date.now();
      const deltaSec = Math.max(0, (now - lastRegenCheck) / 1000);
      lastRegenCheck = now;
      if (energy < maxEnergy) {
        const regenAmount = deltaSec * getEnergyRegenRate();
        energy = Math.min(maxEnergy, energy + regenAmount);
        lastEnergyTimestamp = now;
      } else {
        lastEnergyTimestamp = now;
      }
    }

    function applyAutoTapHit(powerValue, energyCostValue) {
      if (energy < energyCostValue) return;
      if (daily >= dailyCap) return;

      score += powerValue;
      daily += powerValue;
      totalPointsEarned += powerValue;
      checkTotalPointsMilestones();
      addXP(0.2);
      updateEventProgress('tap', 1);
      energy = Math.max(0, energy - energyCostValue);
      registerEnergySpend(energyCostValue);
      dailyClicks++;
      checkTapMilestones();
      dailyPoints += powerValue;
      updateDailyMissionProgress('tap');
      updateDailyMissionProgress('points');
    }

    function runAutoTapTick(deltaMs) {
      if (!isGameInitialized() || !isGameLoopRunning()) return;
      if (!activeBoosts.autoTap.active) return;
      if (playerTier === 'C') {
        activeBoosts.autoTap.active = false;
        stopAutoTapTracking('autoTap');
        return;
      }

      autoTapTickAccumulator += deltaMs;
      while (autoTapTickAccumulator >= 250) {
        autoTapTickAccumulator -= 250;
        applyAutoTapHit(cachedAutoTapPower || (getTapPower() * 0.35), getAutoTapEnergyCost());
      }
    }

    function runSuperAutoTapTick(deltaMs) {
      if (!isGameInitialized() || !isGameLoopRunning()) return;
      if (!activeBoosts.superAutoTap.active) return;
      if (playerTier === 'C') {
        activeBoosts.superAutoTap.active = false;
        stopAutoTapTracking('superAutoTap');
        return;
      }

      superAutoTapTickAccumulator += deltaMs;
      while (superAutoTapTickAccumulator >= 150) {
        superAutoTapTickAccumulator -= 150;
        applyAutoTapHit(cachedSuperAutoTapPower || (getTapPower() * 1.5), getAutoTapEnergyCost());
      }
    }

    function runChallengeTick(deltaMs) {
      challengeTickAccumulator += deltaMs;
      if (challengeTickAccumulator < challengeCheckIntervalMs) return;
      challengeTickAccumulator = 0;

      if (challengeActive) return;
      if (Date.now() < nextChallengeAllowedAt) return;
      const shouldForceSpawn = (Date.now() - lastChallengeShownAt) >= challengeMaxSilenceMs;
      if (!shouldForceSpawn && Math.random() > challengeSpawnChance) return;
      spawnChallenge();
    }

    function handleClaimRequest() {
      if (!requireRunningSession('السحب')) return;
      if (isPermanentlyBanned) {
        showClaimModal("⛔ حظر كامل", "تم حظر السحب نهائيًا بسبب تكرار استخدام البوت الآلي.");
        return;
      }
      if (playerTier === 'C') {
        showClaimModal("غير متاح", "السحب غير متاح حاليًا للفئة C.");
        return;
      }
      if (!userAddress) {
        showClaimModal("❌ خطأ", `يجب ربط المحفظة أولاً قبل سحب ${TOKEN_SYMBOL}`);
        return;
      }

      const now = Date.now();
      const remainingCooldown = Math.max(0, withdrawCooldownMs - (now - lastWithdrawAt));
      if (remainingCooldown > 0) {
        showClaimModal("السحب غير متاح الآن", `يمكنك السحب مرة كل 7 أيام.<br>الوقت المتبقي: <strong>${formatDuration(remainingCooldown)}</strong>`);
        return;
      }

      if (score < pointsPerTapco) {
        showClaimModal("رصيد غير كافي", `تحتاج إلى ${pointsPerTapco} نقطة لسحب 1 ${TOKEN_SYMBOL}.`);
        return;
      }

      const claimAmount = 1;
      const scoreCost = pointsPerTapco;
      const nextWithdrawAt = new Date(now + withdrawCooldownMs);

      showClaimModal(`سحب ${TOKEN_SYMBOL}`, `
        سيتم سحب <strong>${claimAmount.toFixed(2)} ${TOKEN_SYMBOL}</strong><br>
        تكلفة السحب: <strong>${scoreCost} نقطة</strong><br>
        السحب التالي بعد 7 أيام: <strong>${nextWithdrawAt.toLocaleString('ar')}</strong><br><br>
        <small>العقد: ${CONTRACT_ADDRESS}</small><br>
        <small style="color:#94a3b8;">شبكة BSC</small>
      `, true);

      setTimeout(() => {
        score = Math.max(0, score - scoreCost);
        lastWithdrawAt = now;
        savePlayerData();
        updateUI();
        closeClaimModal();
      }, 1500);
    }

    function buyUpgrade(key) {
      if (!requireRunningSession('شراء ترقية')) return;
      const up = upgrades[key];
      const cost = getUpgradeCost(key, up.level + 1);
      if (score < cost) {
        upgradeError.textContent = "❌ نقاطك غير كافية لهذه الترقية.";
        upgradeError.style.display = "block";
        shakeElement('modalBody');
        playSound('error-soft');
        setTimeout(() => upgradeError.style.display = "none", 3000);
        return;
      }
      score -= cost;
      up.level++;
      if (key === 'tapPower') {
        addXP(1); // XP only from tap power upgrade
      }
      checkAchievements('upgrades');
      dailyMissionUpgradesToday++;
      updateDailyMissionProgress('upgrade');

      if (key === "maxEnergy") updateMaxEnergy();
      if (key === "energyRegen") {
        // No extra state needed here; regen speed is calculated dynamically.
      }

      const currentLevel = up.level;
      const upgradeNameMap = {
        tapPower: 'Tap Power',
        maxEnergy: 'Max Energy',
        energyRegen: 'Energy Regen'
      };
      const upgradeLabel = upgradeNameMap[key] || key;

      if (currentLevel >= 20 || cost >= 5000) {
        showBigMoment('🚀 MAJOR UPGRADE', upgradeLabel + ' وصل إلى المستوى ' + currentLevel + '.', '🚀', '#f97316');
        playSound('achievement-big');
      } else if (currentLevel >= 10 || cost >= 1200) {
        showHighlight('Upgrade Purchased: ' + upgradeLabel, '⚡', '#4caf50');
      } else {
        showMinorToast('Upgrade +1: ' + upgradeLabel);
      }

      pulseElement('upgradesBtn');

      updateUI();
      saveUpgrades();

      // === FTUE: First Upgrade Detection ===
      if (!ftueState.firstUpgradeDone) {
        ftueState.firstUpgradeDone = true;
        removeUpgradeArrow();
        showHighlight('قوتك أصبحت أعلى!', '⚡', '#4caf50');
        saveGame();
      }

      renderUpgradesModal();
    }

    function deactivateAllBoosts() {
      Object.keys(activeBoosts).forEach((key) => {
        const boost = activeBoosts[key];
        boost.active = false;
        boost.endTime = 0;

        if (key === 'autoTap') {
          stopAutoTapTracking('autoTap');
        } else if (key === 'superAutoTap') {
          stopAutoTapTracking('superAutoTap');
        }

        if ((key === 'autoTap' || key === 'superAutoTap') && boost.intervalId) {
          clearInterval(boost.intervalId);
          boost.intervalId = null;
        }
      });

      // Defensive reset for any temporary globals that may be introduced later.
      if (typeof window.multiplier !== 'undefined') window.multiplier = 1;
      if (typeof window.xpMultiplier !== 'undefined') window.xpMultiplier = 1;
      if (typeof window.energyCostModifier !== 'undefined') window.energyCostModifier = 1;
      if (typeof window.megaMultiplier !== 'undefined') window.megaMultiplier = 1;
      if (typeof window.autoTapSpeed !== 'undefined') window.autoTapSpeed = 1;

      // Core gameplay values (tap power / XP / energy cost) are computed dynamically from activeBoosts.
      // Turning boosts off above is enough to restore base behavior immediately.
      updateUI();
      updateBoostTimers();
    }

    function buyBoost(type, cost, duration) {
      if (!requireRunningSession('شراء تعزيز')) return;

      const actualCost = (type === 'autoTap')
        ? getUpgradeCost('autoTap', autoTapUses + 1)
        : ((type === 'superAutoTap') ? getUpgradeCost('superAutoTap', superAutoTapUses + 1) : cost);

      // Tier C restriction for auto-tap boosts
      if ((type === 'autoTap' || type === 'superAutoTap') && playerTier === 'C') {
        upgradeError.textContent = "🚫 ميزة النقر التلقائي غير متاحة للفئة C.";
        upgradeError.style.display = "block";
        shakeElement('modalBody');
        playSound('error-soft');
        setTimeout(() => upgradeError.style.display = "none", 3000);
        return;
      }

      if ((type === 'autoTap' || type === 'superAutoTap') && !isGameLoopRunning()) {
        upgradeError.textContent = "⏸️ اللعبة متوقفة حاليًا. شغّل الجلسة أولاً.";
        upgradeError.style.display = "block";
        shakeElement('modalBody');
        playSound('error-soft');
        setTimeout(() => upgradeError.style.display = "none", 3000);
        return;
      }

      if (score < actualCost) {
        upgradeError.textContent = "❌ نقاطك غير كافية لهذا التعزيز.";
        upgradeError.style.display = "block";
        shakeElement('modalBody');
        playSound('error-soft');
        setTimeout(() => upgradeError.style.display = "none", 3000);
        return;
      }
      score -= actualCost;
      totalBoostsUsed++;
      hasEverUsedBoost = true;
      savePlayerData();
      if (type === 'megaMultiplier') hasUsedBoostX5 = true;
      if (['megaMultiplier', 'megaEnergy', 'superAutoTap', 'xpMegaMultiplier'].includes(type)) {
        registerSpecialItemPurchased();
      }

      dailyMissionBoostsUsedToday++;
      if (!dailyMissionDifferentBoostsToday.includes(type)) {
        dailyMissionDifferentBoostsToday.push(type);
      }

      updateUI();

      const now = Date.now();

      deactivateAllBoosts();

      activeBoosts[type].active = true;
      if (duration > 0) {
        activeBoosts[type].endTime = now + duration * 1000;
      }
      if (type === 'autoTap') {
        startAutoTapTracking('autoTap');
        showHighlight('AutoTap Started', '🤖', '#06b6d4');
      } else if (type === 'superAutoTap') {
        startAutoTapTracking('superAutoTap');
        showHighlight('Super AutoTap Started', '🚀', '#ef4444');
      }
      updateDailyMissionProgress('boost');

      // === Instant effects (no duration) ===
      if (type === 'extraEnergy') {
        energy = Math.min(maxEnergy, energy + activeBoosts.extraEnergy.amount);
        activeBoosts.extraEnergy.active = false;
        updateUI();
      } else if (type === 'megaEnergy') {
        energy = Math.min(maxEnergy, energy + activeBoosts.megaEnergy.amount);
        activeBoosts.megaEnergy.active = false;
        updateUI();
      } else if (type === 'instantRecharge') {
        energy = maxEnergy;
        activeBoosts.instantRecharge.active = false;
        updateUI();
      }

      saveAchievements();
      checkAchievements('boosts');
      renderShopModal();
    }

    function executePrestige() {
      if (!canPrestige()) return;

      // Close any open modal
      modalBackdrop.style.display = 'none';

      const prestigeTokens = Math.floor(Math.pow(Math.max(totalPointsEarned, 0), 0.45) / 25);
      prestigeTokensEarned += prestigeTokens;

      // 1. Increment prestige level
      prestigeLevel++;
      registerPrestigeExecution();
      const now = new Date().toISOString();
      if (!prestigeStats.firstPrestigeAt) prestigeStats.firstPrestigeAt = now;
      prestigeStats.lastPrestigeAt = now;

      // 2. Recalculate multipliers
      recalcPrestigeMultipliers();
      if (permanentBonuses.prestigeTokenPurchased) {
        permanentBonuses.pointsMultiplier += 0.05;
        permanentBonuses.xpMultiplier += 0.05;
        permanentBonuses.energyMultiplier += 0.05;
      }

      // 3. RESET: level, xp, upgrades, energy, boosts, challenges
      level = 1;
      xp = 0;

      upgrades.tapPower.level = 0;
      upgrades.maxEnergy.level = 0;
      upgrades.energyRegen.level = 0;
      saveUpgrades();

      // Stop all active boosts
      Object.keys(activeBoosts).forEach(key => {
        const boost = activeBoosts[key];
        if (boost.active) {
          boost.active = false;
          if (boost.intervalId) {
            clearInterval(boost.intervalId);
            boost.intervalId = null;
          }
          boost.endTime = 0;
        }
      });

      // Cancel active challenge
      if (challengeActive) {
        challengeActive = false;
        challengeContainer.style.display = 'none';
        if (challengeHoldTimeout) { clearTimeout(challengeHoldTimeout); challengeHoldTimeout = null; }
      }

      // Reset energy to new base
      updateMaxEnergy();
      energy = maxEnergy;

      // Reset feature unlocks that are level-dependent (keep prestige, specialShop if token purchased)
      unlockedFeatures.autoTap = false;
      unlockedFeatures.advancedChallenges = false;
      unlockedFeatures.newBoosts = false;
      // Keep: prestige (already unlocked), specialShop (keep if was unlocked)

      // 4. PERSIST: score, wallet, referrals, tier, anti-bot, achievements
      // (These are not touched - they remain as-is)

      // 5. Save & update
      savePlayerData();
      updateUI();
      checkAllAchievements();
      updateAchievementsUI();
      checkAchievements('prestige');
      dailyMissionPrestigeToday++;
      updateDailyMissionProgress('prestige');

      // 6. Success message
      const bonusPct = prestigeLevel * 5;
      showBigMoment(
        'PRESTIGE!',
        'تم تنفيذ Prestige بنجاح.\nPrestige Tokens المكتسبة: ' + prestigeTokens.toLocaleString() + '\nمكافآت دائمة: +' + bonusPct + '% نقاط / خبرة / طاقة\nالمستوى التالي: ' + getRequiredLevelForPrestige(),
        '⭐',
        '#f59e0b'
      );
      playSound('prestige-boom');
      pulseElement('prestigeBtn');
    }

    function updateBoostTimers() {
      const now = Date.now();
      ['multiplier', 'extraEnergy', 'autoTap', 'xpMultiplier', 'instantRecharge'].forEach((type, i) => {
        const timerEl = document.getElementById(`timer${i+1}`);

          if (activeBoosts[type].active) {
            const remaining = Math.max(0, Math.floor((activeBoosts[type].endTime - now) / 1000));
            timerEl.textContent = `متبقي: ${remaining} ثانية`;

            timerEl.textContent = '';

        { key: 'xpMegaMultiplier', id: 'spTimer4' }

        const timerEl = document.getElementById(id);

          timerEl.textContent = `متبقي: ${remaining} ثانية`;
        } else {
          timerEl.textContent = '';
        }
      });

      challengeActive = false;
      nextChallengeAllowedAt = Date.now() + challengeCooldownMs;
      challengeContainer.classList.remove('success', 'fail');

        playChallengeSound(outcome);

        clearTimeout(challengeHoldTimeout);
        challengeHoldTimeout = null;
      }
      infoBox.textContent = 'اضغط على زر النقر لزيادة النقاط.';
      const hideDelay = outcome ? 600 : 0;
      if (hideDelay === 0) {
        challengeContainer.style.display = 'none';
      }
      setTimeout(() => {
        challengeContainer.style.display = 'none';
        challengeContainer.classList.remove('success', 'fail');
      }, hideDelay);
    }

      e.stopImmediatePropagation();
      if (!challengeActive || challengeType !== "double") return;

      const now = Date.now();
      challengeTapCount++;

      if (challengeTapCount === 1) {
        lastChallengeTapTime = now;
      } else if (challengeTapCount === 2) {
        if (now - lastChallengeTapTime < 420) {
          score += challengeReward;
          daily += challengeReward;
          totalPointsEarned += challengeReward;
          checkTotalPointsMilestones();
          dailyPoints += challengeReward;
          addXP(3); // XP from successful challenge (economy balance)
          consecutiveChallenges++;
          if (consecutiveChallenges >= 3) {
            addXP(10); // Bonus XP for 3 consecutive challenges
            consecutiveChallenges = 0;
          }
          savePlayerData();
          updateUI();
          const rect = challengeContainer.getBoundingClientRect();
          createFloatingScore(rect.left + rect.width / 2, rect.top + rect.height / 2, challengeReward, true);
          endChallenge('success');
        } else {
          challengeTapCount = 1;
          lastChallengeTapTime = now;
          consecutiveChallenges = 0; // Reset on fail
        }
      }

    function getAchievementProgress(ach) {
      switch (ach.category) {
        case 'points': return totalPointsEarned;
        case 'xp': return level;
        case 'energy': return energySpentTotal;
        case 'upgrades':
          if (ach.subType === 'tapPower') return upgrades.tapPower.level;
          if (ach.subType === 'maxEnergy') return upgrades.maxEnergy.level;
          if (ach.subType === 'energyRegen') return upgrades.energyRegen.level;
          if (ach.subType === 'any') return Math.max(upgrades.tapPower.level, upgrades.maxEnergy.level, upgrades.energyRegen.level);
          return 0;
        case 'boosts':
          if (ach.subType === 'total') return totalBoostsUsed;
          if (ach.subType === 'autoTap') return autoTapUses;
          if (ach.subType === 'autoTap_time') return autoTapTotalTime;
          if (ach.subType === 'superAutoTap') return superAutoTapUses;
          if (ach.subType === 'superAutoTap_time') return superAutoTapTotalTime;
          if (ach.subType === 'boostX5') return hasUsedBoostX5 ? 1 : 0;
          return 0;
        case 'prestige':
          if (ach.subType === 'count') return prestigeCount;
          if (ach.subType === 'tokens') return prestigeTokensEarned;
          return prestigeLevel;
        case 'referrals': return referralCount;
        case 'secret':
          if (ach.subType === 'dailyClicks') return dailyClicks;
          if (ach.subType === 'noBoosts') return (!hasEverUsedBoost && level >= 50) ? 1 : 0;
          if (ach.subType === 'specialShop') return hasOpenedSpecialShop ? 1 : 0;
          if (ach.subType === 'dailyPoints') return dailyPoints;
          if (ach.subType === 'noReferrals') return (totalPointsEarned >= 100000000 && referralCount === 0) ? 1 : 0;
          return 0;
        default: return 0;
      }
    }

    function checkAchievements(category) {
      if (playerTier === 'C' && (category === 'points' || category === 'xp')) return;
      if (playerTier === 'C' && category === 'boosts' && activeBoosts.autoTap.active) return;
      achievementsList.forEach(ach => {
        if (category !== 'all' && ach.category !== category) return;
        if (completedAchievements.includes(ach.id)) return;
        const progress = getAchievementProgress(ach);
        if (progress >= ach.targetValue) {
          unlockAchievement(ach);
        }
      });
    }

    function checkAllAchievements() {
      const autoTapUseMet = autoTapUses >= 1;
      const autoTapTimeMet = autoTapTotalTime >= 60000;
      const superAutoTapUseMet = superAutoTapUses >= 1;
      const superAutoTapTimeMet = superAutoTapTotalTime >= 60000;
      const prestigeCountMilestonesMet = (prestigeCount >= 1 || prestigeCount >= 5 || prestigeCount >= 10);
      const prestigeTokensMilestonesMet = (prestigeTokensEarned >= 1 || prestigeTokensEarned >= 10);
      const energyMilestonesMet = (energySpentTotal >= 1000 || energySpentTotal >= 5000 || energySpentTotal >= 10000);
      const metaAchievementMilestonesMet = (unlockedAchievementsCount >= 5 || unlockedAchievementsCount >= 10 || unlockedAchievementsCount >= 20);
      const metaSecretMilestonesMet = (unlockedSecretAchievementsCount >= 1 || unlockedSecretAchievementsCount >= 3 || unlockedSecretAchievementsCount >= 5);
      const externalSystemMilestonesMet = (
        hasOpenedSpecialShop ||
        specialItemPurchasedCount >= 1 ||
        eventsJoinedCount >= 1 ||
        referralCount >= 1 ||
        Object.keys(specialTabsOpened).length >= 1
      );
      const categoryCompletionMetaMet = ['points', 'xp', 'energy', 'upgrades', 'boosts', 'prestige', 'referrals'].some(function(category) {
        const categoryAchievements = achievementsList.filter(function(ach) { return ach.category === category; });
        if (categoryAchievements.length === 0) return false;
        return categoryAchievements.every(function(ach) { return completedAchievements.includes(ach.id); });
      });

      ['points', 'xp', 'energy', 'upgrades', 'boosts', 'prestige', 'referrals'].forEach(cat => checkAchievements(cat));

      if (autoTapUseMet || autoTapTimeMet || superAutoTapUseMet || superAutoTapTimeMet) {
        checkAchievements('boosts');
      }
      if (energyMilestonesMet) {
        checkAchievements('energy');
      }
      if (prestigeCountMilestonesMet || prestigeTokensMilestonesMet) {
        checkAchievements('prestige');
      }
      if (metaAchievementMilestonesMet || metaSecretMilestonesMet || categoryCompletionMetaMet) {
        checkAchievements('secret');
      }
      if (externalSystemMilestonesMet) {
        checkAchievements('secret');
        checkAchievements('referrals');
      }

      achievementsList.forEach(ach => {
        if (ach.category !== 'secret') return;
        if (completedAchievements.includes(ach.id)) return;
        if (ach.subType === 'noBoosts' && hasEverUsedBoost) return;

        const progress = getAchievementProgress(ach);
        if (progress >= ach.targetValue) {
          unlockAchievement(ach);
        }
      });
    }

    function registerAchievementUnlocked(achievementId) {
      if (completedAchievements.includes(achievementId)) return false;

      completedAchievements.push(achievementId);
      achievementUnlockTimestamps[achievementId] = Date.now();

      unlockedAchievementsCount = completedAchievements.length;
      unlockedSecretAchievementsCount = completedAchievements.filter(function(id) {
        const ach = achievementsList.find(function(a) { return a.id === id; });
        return !!(ach && ach.isSecret);
      }).length;

      saveAchievements();
      savePlayerData();
      updateAchievementsUI();

      if (!achievementMetaSyncInProgress) {
        achievementMetaSyncInProgress = true;
        try {
          checkAllAchievements();
        } finally {
          achievementMetaSyncInProgress = false;
        }
      }

      return true;
    }

    function isMetaAchievement(ach) {
      if (!ach || ach.isSecret) return false;
      if (ach.rewardType === 'badge') return true;
      if (ach.rewardType === 'permanent_points' || ach.rewardType === 'permanent_xp' || ach.rewardType === 'permanent_energy') return true;
      return false;
    }

    function getCompletedAchievementCategories() {
      const result = [];
      ['points', 'xp', 'energy', 'upgrades', 'boosts', 'prestige', 'referrals'].forEach(function(category) {
        const categoryAchievements = achievementsList.filter(function(item) { return item.category === category; });
        if (!categoryAchievements.length) return;
        if (categoryAchievements.every(function(item) { return completedAchievements.includes(item.id); })) {
          result.push(category);
        }
      });
      return result;
    }

    function unlockAchievement(ach) {
      const categoriesBefore = getCompletedAchievementCategories();
      if (!registerAchievementUnlocked(ach.id)) return;
      dailyMissionAchievementsToday++;
      updateDailyMissionProgress('achievement');

      const typeTapPowerBonus = ach.isSecret ? 0.10 : (isMetaAchievement(ach) ? 0.05 : 0.02);
      achievementBonuses.tapPowerPercent += typeTapPowerBonus;

      if (ach.badge && !achievementBadges.includes(ach.badge)) {
        achievementBadges.push(ach.badge);
      }
      saveAchievements();
      savePlayerData();
      updateAchievementsUI();

      const categoriesAfter = getCompletedAchievementCategories();
      const newlyCompletedCategory = categoriesAfter.find(function(cat) { return !categoriesBefore.includes(cat); });

      if (newlyCompletedCategory) {
        achievementBonuses.tapPowerPercent += 0.15;
      }

      saveAchievements();
      savePlayerData();

      if (newlyCompletedCategory) {
        showBigMoment('🏁 CATEGORY COMPLETED', 'أكملت فئة إنجازات كاملة: ' + newlyCompletedCategory, '🏁', '#22c55e');
        playSound('achievement-big');
      } else if (ach.isSecret) {
        showBigMoment('🕵️ SECRET ACHIEVEMENT', ach.name + '\n' + ach.description, '🕵️', '#00e5ff');
        playSound('achievement-big');
      } else if (isMetaAchievement(ach)) {
        showBigMoment('🌟 META ACHIEVEMENT', ach.name + '\n' + ach.description, '🌟', '#ffd700');
        playSound('achievement-big');
      } else {
        showHighlight('Achievement Unlocked: ' + ach.name, '⭐', '#ffd700');
      }

      if (completedAchievements.length >= 5 && markFeedbackMilestone('achievements_5')) {
        showHighlight('Milestone: 5 Achievements!', '🏆', '#f59e0b');
        flashCounter('achievementCounter');

        flashCounter('achievementCounter');

        flashCounter('achievementCounter');
      }

      flashCounter('achievementCounter');
    }

    function getUTCMidnight() {
      const now = new Date();
      const utcMidnight = new Date(Date.UTC(now.getUTCFullYear(), now.getUTCMonth(), now.getUTCDate()));
      return utcMidnight.getTime();
    }

    function getNextUTCMidnight() {
      return getUTCMidnight() + 86400000;
    }

    function getDailyResetCountdown() {
      const diff = getNextUTCMidnight() - Date.now();
      if (diff <= 0) return '00:00:00';
      const h = Math.floor(diff / 3600000);
      const m = Math.floor((diff % 3600000) / 60000);
      const s = Math.floor((diff % 60000) / 1000);
      return (h < 10 ? '0' : '') + h + ':' + (m < 10 ? '0' : '') + m + ':' + (s < 10 ? '0' : '') + s;
    }

    function checkDailyMissionReset() {
      const todayMidnight = getUTCMidnight();

        // New day - reset

        dailyMissionCompletedCount = 0;
        dailyBonusClaimed = false;

    function generateDailyMissions() {
      const easyPool = dailyMissionsList.filter(function(m) { return m.difficulty === 'easy'; });
      const mediumPool = dailyMissionsList.filter(function(m) { return m.difficulty === 'medium'; });
      const hardPool = dailyMissionsList.filter(function(m) { return m.difficulty === 'hard'; });

      function pickRandom(arr) {
        return arr[Math.floor(Math.random() * arr.length)];
      }

      const easyMission = pickRandom(easyPool);
      const mediumMission = pickRandom(mediumPool);
      const hardMission = pickRandom(hardPool);

      function assignReward(difficulty) {
        const options = dailyMissionRewards[difficulty];
        return options[Math.floor(Math.random() * options.length)];
      }

      // 5% chance of rare badge for hard missions
      let hardReward = assignReward('hard');
      if (Math.random() < 0.05) {
        hardReward = { type: 'badge', value: 0, badgeName: 'Daily Champion' };
      }

      activeDailyMissions = [
        { ...easyMission, progress: 0, completed: false, claimed: false, reward: assignReward('easy') },
        { ...mediumMission, progress: 0, completed: false, claimed: false, reward: assignReward('medium') },
        { ...hardMission, progress: 0, completed: false, claimed: false, reward: hardReward }
      ];
    }

    function getDailyMissionCurrentValue(mission) {
      switch (mission.type) {
        case 'points': return dailyPoints;
        case 'xp': return dailyXP;
        case 'energy': return dailyEnergyConsumed;
        case 'tap': return dailyClicks;
        case 'boost': return dailyMissionBoostsUsedToday;
        case 'boost_auto': return 0; // tracked via progress
        case 'boost_x5': return 0; // tracked via progress
        case 'upgrade': return dailyMissionUpgradesToday;
        case 'prestige': return dailyMissionPrestigeToday;
        case 'prestige_token': return dailyMissionPrestigeToday; // same trigger
        case 'special_shop': return hasOpenedSpecialShop ? 1 : 0;
        case 'achievement': return dailyMissionAchievementsToday;

        case 'hourly_points': return dailyMissionHourlyPoints;

        default: return 0;
      }
    }

    function updateDailyMissionProgress(triggerType) {
      // Anti-cheat: no progress for banned/tier-C/tier-locked players
      if (playerTier === 'C') return;
      if (typeof isBanned !== 'undefined' && isBanned) return;

      if (!activeDailyMissions || activeDailyMissions.length === 0) return;

      // Track hourly points
      if (triggerType === 'points') {
        const now = Date.now();
        if (dailyMissionHourlyStart === 0) {
          dailyMissionHourlyStart = now;
          dailyPointsAtHourStart = dailyPoints;
        }
        if (now - dailyMissionHourlyStart > 3600000) {
          dailyMissionHourlyStart = now;
          dailyPointsAtHourStart = dailyPoints;
        }
        dailyMissionHourlyPoints = Math.max(0, dailyPoints - dailyPointsAtHourStart);
      }

      activeDailyMissions.forEach(function(mission) {
        if (mission.completed) return;

        let shouldUpdate = false;
        let currentVal = 0;

        switch (mission.type) {
          case 'points':
            if (triggerType === 'points') { shouldUpdate = true; currentVal = dailyPoints; }
            break;
          case 'xp':
            if (triggerType === 'xp') { shouldUpdate = true; currentVal = dailyXP; }
            break;
          case 'energy':
            if (triggerType === 'energy') {
              shouldUpdate = true;
              currentVal = dailyEnergyConsumed;
            }
            break;
          case 'tap':
            if (triggerType === 'tap') { shouldUpdate = true; currentVal = dailyClicks; }
            break;
          case 'boost':
            if (triggerType === 'boost') { shouldUpdate = true; currentVal = dailyMissionBoostsUsedToday; }
            break;
          case 'boost_auto':
            if (triggerType === 'boost' && (activeBoosts.autoTap.active || activeBoosts.superAutoTap.active)) {
              shouldUpdate = true; currentVal = 1;
            }
            break;
          case 'boost_x5':
            if (triggerType === 'boost' && activeBoosts.megaMultiplier.active) {
              shouldUpdate = true; currentVal = 1;
            }
            break;
          case 'upgrade':
            if (triggerType === 'upgrade') { shouldUpdate = true; currentVal = dailyMissionUpgradesToday; }
            break;
          case 'prestige':
          case 'prestige_token':
            if (triggerType === 'prestige') { shouldUpdate = true; currentVal = dailyMissionPrestigeToday; }
            break;
          case 'special_shop':
            if (hasOpenedSpecialShop) { shouldUpdate = true; currentVal = 1; }
            break;
          case 'achievement':
            if (triggerType === 'achievement') { shouldUpdate = true; currentVal = dailyMissionAchievementsToday; }
            break;
          case 'consecutive_energy':
            if (triggerType === 'energy') { shouldUpdate = true; currentVal = dailyMissionConsecutiveEnergy; }
            break;
          case 'hourly_points':
            if (triggerType === 'points') { shouldUpdate = true; currentVal = dailyMissionHourlyPoints; }
            break;
          case 'different_boosts':
            if (triggerType === 'boost') { shouldUpdate = true; currentVal = dailyMissionDifferentBoostsToday.length; }
            break;
        }

        if (shouldUpdate) {
          mission.progress = Math.max(mission.progress, currentVal);
          if (mission.progress >= mission.target && !mission.completed) {
            mission.completed = true;
            // === DAILY EXPERIENCE: Mission Complete Feedback ===
            showHighlight('مهمة يومية مكتملة! ⭐', '✨', '#4caf50');
          }
        }
      });
      saveDailyMissions();
    }

    }

    function startGameLoop() {
      if (!isGameInitialized() || isGameLoopRunning()) return;

      lifecycleState = 'running';
      lifecycleLoopRunning = true;
      lastRegenCheck = Date.now();

      gameLoopCounter = 0;
      autoTapTickAccumulator = 0;
      superAutoTapTickAccumulator = 0;
      refreshEconomyCache();

      gameLoopIntervalId = setInterval(function() {
        if (!isGameLoopRunning()) return;

        gameLoopCounter++;
        tickEnergyRegen();
        runAutoTapTick(100);
        runSuperAutoTapTick(100);

        if (gameLoopCounter % 10 === 0) {
          refreshEconomyCache();
          tickEventSystems();
          tickBehaviorAndBonuses();
          runChallengeTick(1000);
        }

        if (gameLoopCounter % 3 === 0) {
          tickUiRefresh();
        }

        if (gameLoopCounter % 600 === 0) {
          checkDailyMissionReset();
          tickMinuteSystems();
          const dmCounter = document.getElementById('dailyMissionCounter');
          if (dmCounter) updateTextIfChanged(dmCounter, dailyMissionCompletedCount + '/3');
        }
      }, 100);

      startChallengeSystem();
    }

    function stopGameLoop() {
      if (!isGameLoopRunning()) return;

      lifecycleLoopRunning = false;
      lifecycleState = 'stopped';

      if (gameLoopIntervalId) {
        clearInterval(gameLoopIntervalId);
        gameLoopIntervalId = null;
      }

      clearLifecycleIntervals();

      if (dailyMissionCountdownInterval) {
        clearInterval(dailyMissionCountdownInterval);
        dailyMissionCountdownInterval = null;
      }

      if (challengeHoldTimeout) {
        clearTimeout(challengeHoldTimeout);
        challengeHoldTimeout = null;
      }

      if (activeBoosts.autoTap.intervalId) {
        clearInterval(activeBoosts.autoTap.intervalId);
        activeBoosts.autoTap.intervalId = null;
      }
      if (activeBoosts.superAutoTap.intervalId) {
        clearInterval(activeBoosts.superAutoTap.intervalId);
        activeBoosts.superAutoTap.intervalId = null;
      }
    }

    function shutdownGame() {

      saveGame();
      lifecycleState = 'shutdown';

    function isNewDay() {
      const today = new Date().toDateString();
      return dailyState.lastLoginDate !== today;
    }

    function updateDailyStreak() {
      const today = new Date().toDateString();

      if (!dailyState.lastLoginDate) {
        dailyState.streak = 1;
      } else {
        const lastDate = new Date(dailyState.lastLoginDate);
        const todayDate = new Date();
        const diffMs = todayDate.getTime() - lastDate.getTime();
        const diffDays = diffMs / (1000 * 60 * 60 * 24);

        if (diffDays < 2) {
          dailyState.streak++;
        } else {
          dailyState.streak = 1;
        }
      }

      dailyState.lastLoginDate = today;
      saveGame();
      return dailyState.streak;
    }

    function applyDailyBoost(streak) {
      let boost = 0;
      let duration = 0;

      if (streak >= 1) { boost = 0.10; duration = 600; }
      if (streak >= 2) { boost = 0.15; duration = 600; }
      if (streak >= 3) { boost = 0.20; duration = 900; }
      if (streak >= 7) { boost = 0.30; duration = 1200; }
      if (streak >= 14) { boost = 0.40; duration = 1500; }
      if (streak >= 30) { boost = 0.50; duration = 1800; }

      if (boost > 0) {
        dailyState.dailyBoost = { 
          active: true, 
          boost: boost, 
          duration: duration, 
          startTime: Date.now() 
        };
        const mins = Math.ceil(duration / 60);
        showBigMoment('⚡ Boost يومي!', `قوتك +${Math.round(boost * 100)}% لمدة ${mins} دقيقة`, '⚡', '#ff9800');
        playSound('achievement-soft');
      }
    }

    function updateDailyBoostMultiplier() {
      if (!dailyState.dailyBoost.active) return 1;
      
      const elapsed = (Date.now() - dailyState.dailyBoost.startTime) / 1000;
      if (elapsed > dailyState.dailyBoost.duration) {
        dailyState.dailyBoost.active = false;
        return 1;
      }
      
      return 1 + dailyState.dailyBoost.boost;
    }

    function updateTempBoostMultiplier() {
      if (!dailyState.tempBoost.active) return 1;
      
      const elapsed = (Date.now() - dailyState.tempBoost.startTime) / 1000;
      if (elapsed > dailyState.tempBoost.duration) {
        dailyState.tempBoost.active = false;
        return 1;
      }
      
      if (dailyState.tempBoost.type === 'tap') {
        return dailyState.tempBoost.value;
      }
      
      return 1;
    }

    function processToastQueue() {
      if (!canShowPlayerFeedback()) {
        feedbackToastQueue = [];
        return;
      }
      if (feedbackBigMomentActive) return;
      if (!toast || feedbackToastQueue.length === 0) return;

      const now = Date.now();
      if (now < feedbackToastBusyUntil) {
        setTimeout(processToastQueue, Math.max(60, feedbackToastBusyUntil - now));
        return;
      }

      const nextToast = feedbackToastQueue.shift();
      if (!nextToast) return;
      showToast(nextToast.message, nextToast.duration || 1200, true);
    }

    function renderHighlight(message, icon, color) {
      const el = document.createElement('div');
      el.className = 'feedback-highlight';
      el.style.borderColor = color || '#4caf50';
      el.style.boxShadow = '0 10px 30px rgba(2, 6, 23, 0.65), 0 0 24px ' + ((color || '#4caf50') + '55');
      el.innerHTML =
        '<span class="feedback-highlight-icon">' + (icon || '✨') + '</span>' +
        '<div class="feedback-highlight-text">' + (message || '') + '</div>';
      document.body.appendChild(el);
      playSound('achievement-soft');
      setTimeout(function() {
        el.remove();
      }, 2000);
    }

    function processHighlightQueue() {
      if (!canShowPlayerFeedback()) {
        feedbackHighlightQueue = [];
        return;
      }
      if (feedbackBigMomentActive) return;
      if (!feedbackHighlightQueue.length) return;

      const now = Date.now();
      if (now < feedbackHighlightBusyUntil) {
        setTimeout(processHighlightQueue, Math.max(80, feedbackHighlightBusyUntil - now));
        return;
      }

      const nextHighlight = feedbackHighlightQueue.shift();
      if (!nextHighlight) return;
      feedbackHighlightBusyUntil = Date.now() + 1900;
      renderHighlight(nextHighlight.message, nextHighlight.icon, nextHighlight.color);

      if (feedbackHighlightQueue.length) {
        setTimeout(processHighlightQueue, 1900);
      }
    }

    function showToast(message, duration = 1200, immediate = false) {
      if (!canShowPlayerFeedback()) return;
      if (!toast) return;

      if (!immediate) {
        feedbackToastQueue.push({ message: String(message || ''), duration: Math.max(200, duration) });
        processToastQueue();
        return;
      }

      toast.textContent = String(message || '');
      toast.classList.add('show');
      if (feedbackToastTimeout) clearTimeout(feedbackToastTimeout);
      feedbackToastBusyUntil = Date.now() + Math.max(240, duration + 180);
      feedbackToastTimeout = setTimeout(function() {
        toast.classList.remove('show');
        toast.textContent = '';
        setTimeout(processToastQueue, 120);
      }, Math.min(1500, Math.max(200, duration)));
    }

    function showMinorToast(message, duration = 1000, cooldownMs = 700) {
      if (!canShowPlayerFeedback()) return;

      const safeMessage = String(message || '').trim();
      if (!safeMessage) return;

      const now = Date.now();
      if (feedbackLastMinorToast.message === safeMessage && now - feedbackLastMinorToast.at < cooldownMs) {
        return;
      }

      feedbackLastMinorToast = { message: safeMessage, at: now };
      showToast(safeMessage, duration);
    }

    function playSound(name) {
      if (!canShowPlayerFeedback()) return;
      switch (name) {
        case 'click-soft':
          playTone(420, 55, 'sine', 0.05);
          break;
        case 'click-medium':
          playTone(560, 80, 'triangle', 0.08);
          break;
        case 'achievement-soft':
          playTone(760, 110, 'sine', 0.14);
          setTimeout(function() { playTone(980, 90, 'sine', 0.1); }, 90);
          break;
        case 'achievement-big':
          playTone(720, 160, 'triangle', 0.16);
          setTimeout(function() { playTone(980, 140, 'sine', 0.16); }, 120);
          setTimeout(function() { playTone(1240, 190, 'sine', 0.18); }, 250);
          break;
        case 'prestige-boom':
          playTone(420, 130, 'square', 0.14);
          setTimeout(function() { playTone(760, 180, 'triangle', 0.18); }, 100);
          setTimeout(function() { playTone(1180, 220, 'sine', 0.22); }, 260);
          break;
        case 'error-soft':
          playTone(220, 90, 'square', 0.1);
          break;
        default:
          break;
      }
    }

    function showHighlight(message, icon, color) {
      if (!canShowPlayerFeedback()) return;
      feedbackHighlightQueue.push({

        icon: icon || '✨',
        color: color || '#4caf50'
      });

      if (feedbackHighlightQueue.length > 12) {
        feedbackHighlightQueue = feedbackHighlightQueue.slice(-12);
      }

      processHighlightQueue();
    }

      if (!canShowPlayerFeedback()) return;

      if (existing) existing.remove();

    function getReferralTapBonusPercentForSimulation(referrals) {
      const cappedReferrals = Math.max(0, Math.min(20, parseInt(referrals || 0, 10)));
      if (cappedReferrals === 0) return 0;
      return 0.05 + Math.max(0, cappedReferrals - 1) * 0.01;
    }

    function getAchievementTapBonusPercentForSimulation(config) {
      const safeConfig = config || {};
      const normal = Math.max(0, parseInt(safeConfig.normal || 0, 10));
      const meta = Math.max(0, parseInt(safeConfig.meta || 0, 10));
      const secret = Math.max(0, parseInt(safeConfig.secret || 0, 10));
      const category = Math.max(0, parseInt(safeConfig.category || 0, 10));
      return normal * 0.02 + meta * 0.05 + secret * 0.10 + category * 0.15;
    }

    function simulateEconomyProfile(profile) {
      const p = profile || {};
      const levelSim = Math.max(1, parseInt(p.level || 1, 10));
      const tapUpgradeLevel = Math.max(0, parseInt(p.tapUpgradeLevel || 0, 10));
      const maxEnergyUpgradeLevel = Math.max(0, parseInt(p.maxEnergyUpgradeLevel || 0, 10));
      const regenUpgradeLevel = Math.max(0, parseInt(p.regenUpgradeLevel || 0, 10));
      const prestigeLevelSim = Math.max(0, parseInt(p.prestigeLevel || 0, 10));
      const referrals = Math.max(0, parseInt(p.referrals || 0, 10));
      const totalPoints = Math.max(0, Number(p.totalPointsEarned || 0));
      const achievementBonus = getAchievementTapBonusPercentForSimulation(p.achievements || {});
      const referralBonus = getReferralTapBonusPercentForSimulation(referrals);
      const prestigeMultiplier = 1 + (prestigeLevelSim * 0.05);

      const tapPower = economyBaseTap * Math.pow(economyTapGrowthRate, tapUpgradeLevel) * (1 + achievementBonus + referralBonus) * prestigeMultiplier;
      const autoTapPower = tapPower * 0.35;
      const superAutoTapPower = tapPower * 1.5;

      const maxEnergySim = 100 + (maxEnergyUpgradeLevel * 12);
      const regenPerSec = 1 + (regenUpgradeLevel * economyEnergyRegenGrowthRate);
      const autoEnergyCostSim = getAutoTapEnergyCostForLevel(levelSim);
      const manualCps = 4;
      const autoCps = 4;
      const superCps = 6.67;

      function getDrainSeconds(consumptionPerSec) {
        const net = regenPerSec - consumptionPerSec;
        if (net >= 0) return '∞';
        return Math.floor(maxEnergySim / Math.abs(net));
      }

      const prestigeTokensEstimate = Math.floor(Math.pow(Math.max(totalPoints, 0), 0.45) / 25);
      const dailyRewardSim = Math.floor((100 + (levelSim * 15) + (Math.sqrt(Math.max(1, tapPower)) * 35)) * (1 + prestigeLevelSim * 0.02));
      const eventRewardEasy = tapPower * (50 + 1 * 20);
      const eventRewardMedium = tapPower * (50 + 2 * 20);
      const eventRewardHard = tapPower * (50 + 3 * 20);

      return {
        scenario: p.name || 'profile',
        level: levelSim,
        tapPower: Math.round(tapPower * 100) / 100,
        autoTapPower: Math.round(autoTapPower * 100) / 100,
        superAutoTapPower: Math.round(superAutoTapPower * 100) / 100,
        maxEnergy: maxEnergySim,
        regenPerSec: Math.round(regenPerSec * 100) / 100,
        manualDrainSec: getDrainSeconds(tapEnergyCost * manualCps),
        autoDrainSec: getDrainSeconds(autoEnergyCostSim * autoCps),
        superDrainSec: getDrainSeconds(autoEnergyCostSim * superCps),
        dailyReward: Math.floor(dailyRewardSim),
        eventEasy: Math.floor(eventRewardEasy),
        eventMedium: Math.floor(eventRewardMedium),
        eventHard: Math.floor(eventRewardHard),
        nextTapUpgradeCost: getUpgradeCost('tapPower', tapUpgradeLevel + 1),
        nextEnergyUpgradeCost: getUpgradeCost('maxEnergy', maxEnergyUpgradeLevel + 1),
        nextRegenUpgradeCost: getUpgradeCost('energyRegen', regenUpgradeLevel + 1),
        nextAutoTapCost: getUpgradeCost('autoTap', Math.max(1, parseInt(p.autoTapUses || 0, 10) + 1)),
        nextSuperAutoTapCost: getUpgradeCost('superAutoTap', Math.max(1, parseInt(p.superAutoTapUses || 0, 10) + 1)),
        prestigeTokensEstimate: prestigeTokensEstimate,
        referralBonusPercent: Math.round(referralBonus * 100),
        achievementTapBonusPercent: Math.round(achievementBonus * 100)
      };
    }
