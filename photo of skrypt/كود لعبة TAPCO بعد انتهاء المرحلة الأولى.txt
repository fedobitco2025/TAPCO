      scoreEl.textContent = Math.floor(score);
      xpText.textContent = `XP: ${xp} | المستوى: ${level}`;
      dailyText.textContent = `اليومي: ${Math.floor(daily)} / ${dailyCap}`;

      energyTextEl.textContent = `${Math.floor(energy)} / ${maxEnergy}`;
    }

    function loadUpgrades() {
      const saved = localStorage.getItem("tapco_upgrades");
      if (saved) upgrades = JSON.parse(saved);
      maxEnergy = 100 + upgrades.maxEnergy.level * upgrades.maxEnergy.baseBonus;
      energyRegenRate = 1 + upgrades.energyRegen.level * upgrades.energyRegen.baseBonus;
    }

    function saveUpgrades() {
      localStorage.setItem("tapco_upgrades", JSON.stringify(upgrades));
    }

    function getTapPower() {
      return 0.1 + (upgrades.tapPower.level * upgrades.tapPower.baseBonus);
    }

    function createFloatingScore(x, y, amount, isBonus = false) {
      const floating = document.createElement('div');
      floating.className = 'floating-score';
      floating.textContent = isBonus ? `+${amount} 🎁` : `+${amount.toFixed(1)}`;
      floating.style.left = `${x}px`;
      floating.style.top = `${y}px`;
      if (isBonus) floating.style.color = '#22c55e';
      tapArea.appendChild(floating);
      setTimeout(() => floating.remove(), 1500);
    }

    tapBtn.addEventListener('click', (e) => {
      if (challengeActive) return;
      if (Date.now() - lastTapTime < tapCooldownMs) return;
      lastTapTime = Date.now();

      if (daily >= dailyCap || energy <= 0) return;

      const power = getTapPower() * (1 + (level - 1) * 0.01);
      score += power;
      daily += power;
      xp += 1;
      energy = Math.max(0, energy - 2);

      if (xp >= 100) { xp -= 100; level++; }

      updateUI();

      const rect = tapArea.getBoundingClientRect();
      let clickX = e.clientX ? e.clientX - rect.left : rect.width / 2;
      let clickY = e.clientY ? e.clientY - rect.top - 45 : rect.height / 2 - 45;
      createFloatingScore(clickX, clickY, power);

      // Create ripple effect
      const ripple = document.createElement('div');
      ripple.className = 'ripple';
      ripple.style.left = `${clickX - 10}px`;
      ripple.style.top = `${clickY - 10}px`;
      tapArea.appendChild(ripple);
      setTimeout(() => ripple.remove(), 600);

      if (navigator.vibrate) navigator.vibrate(12);
    });

    setInterval(() => {
      if (energy < maxEnergy) {
        energy = Math.min(maxEnergy, energy + energyRegenRate);
        updateUI();
      }
    }, 3000);

    function buyUpgrade(key) {
      const up = upgrades[key];
      if (score < up.cost) {
        upgradeError.textContent = "❌ نقاطك غير كافية لهذه الترقية.";
        upgradeError.style.display = "block";
        setTimeout(() => upgradeError.style.display = "none", 3000);
        return;
      }
      score -= up.cost;
      up.level++;
      up.cost = Math.floor(up.cost * 1.8);

      if (key === "maxEnergy") maxEnergy = 100 + upgrades.maxEnergy.level * 20;
      if (key === "energyRegen") energyRegenRate = 1 + upgrades.energyRegen.level * 1;

    function spawnChallenge() {
      if (challengeActive) return;
      challengeActive = true;
      challengeTapCount = 0;
      lastChallengeTapTime = 0;

      const types = [
        { type: "double", icon: "⚡", label: "اضغط مرتين بسرعة!", reward: 10 },
        { type: "hold",   icon: "🔥", label: "اضغط مطولاً 3 ثواني!", reward: 12 }
      ];
      const selected = types[Math.floor(Math.random() * types.length)];

      challengeType = selected.type;
      challengeReward = selected.reward;
      challengeIcon.textContent = selected.icon;
      challengeLabel.textContent = selected.label;

      const containerRect = tapArea.getBoundingClientRect();
      const maxX = containerRect.width - 80;
      const maxY = containerRect.height - 80;
      const randomX = 30 + Math.random() * maxX;
      const randomY = 20 + Math.random() * maxY;

      challengeContainer.style.left = `${randomX}px`;
      challengeContainer.style.top = `${randomY}px`;
      challengeContainer.style.display = 'flex';

      tapBtn.classList.add('disabled');
    }

    function hideChallenge() {
      challengeActive = false;
      challengeContainer.style.display = 'none';
      tapBtn.classList.remove('disabled');
      if (challengeHoldTimeout) clearTimeout(challengeHoldTimeout);
    }

    challengeContainer.addEventListener('click', (e) => {
      e.stopImmediatePropagation();
      if (!challengeActive || challengeType !== "double") return;

      const now = Date.now();
      challengeTapCount++;

      if (challengeTapCount === 1) {
        lastChallengeTapTime = now;
      } else if (challengeTapCount === 2) {
        if (now - lastChallengeTapTime < 420) {
          score += challengeReward;
          daily += challengeReward;
          updateUI();
          const rect = challengeContainer.getBoundingClientRect();
          const areaRect = tapArea.getBoundingClientRect();
          createFloatingScore(rect.left - areaRect.left + 27, rect.top - areaRect.top + 27, challengeReward, true);
          hideChallenge();
        } else {
          challengeTapCount = 1;
          lastChallengeTapTime = now;
        }
      }
    });

    challengeContainer.addEventListener('pointerdown', (e) => {
      if (!challengeActive || challengeType !== "hold") return;
      e.stopImmediatePropagation();
      challengeHoldTimeout = setTimeout(() => {
        if (challengeActive) {
          score += challengeReward;
          daily += challengeReward;
          updateUI();
          const rect = challengeContainer.getBoundingClientRect();
          const areaRect = tapArea.getBoundingClientRect();
          createFloatingScore(rect.left - areaRect.left + 27, rect.top - areaRect.top + 27, challengeReward, true);
          hideChallenge();
        }
      }, 3000);
    });

    challengeContainer.addEventListener('pointerup', () => {
      if (challengeHoldTimeout) {
        clearTimeout(challengeHoldTimeout);
        challengeHoldTimeout = null;
      }
    });

    function startChallengeSystem() {
      setTimeout(() => { if (!challengeActive) spawnChallenge(); }, 12000);
      setInterval(() => {
        if (!challengeActive) spawnChallenge();
      }, 55000);
    }

    function init() {
      loadUpgrades();
      updateUI();
      updateLeaderboard();
      startChallengeSystem();

    }
    init();
  </script>
